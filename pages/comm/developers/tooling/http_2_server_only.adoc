== Run

npm install http2 -g

echo "require('http2').createServer({}, function(request, response) {
response.end('Hello world!');
}).listen(8080);" | tee /tmp/server.js > /dev/null

node /tmp/server.js

== Test

curl http://localhost:8080/ -v --http2-prior-knowledge
